            if (lower.includes('modern') || lower.includes('mt') || lower.includes('retail') || lower.includes('store')) return 'Modern trade';

            if (lower.includes('online') || lower.includes('web') || lower.includes('lazada') || lower.includes('shopee')) return 'Online';

            if (lower.includes('event') || lower.includes('fair') || lower.includes('booth') || lower.includes('exhibition')) return 'Event';

            if (lower.includes('b2b') || lower.includes('project') || lower.includes('corporate')) return 'B2B Project';

            if (lower.includes('sub') || lower.includes('rent')) return 'Subscription';

            return 'Other';

        const enrichData = (obj) => {

            let qty = parseFloat(obj.Qty) || 0;

            let revenueIncVat = parseFloat(obj.UnitPriceIncVat) || 0;

            let unitPrice = parseFloat(obj.UnitPrice) || 0;

            let cost = parseFloat(obj.AvgCost) || 0;

            

            if (obj.InvType === 'C' || obj.InvType === 'CN') {

                qty = -Math.abs(qty);

                revenueIncVat = -Math.abs(revenueIncVat);

                unitPrice = -Math.abs(unitPrice);

                cost = -Math.abs(cost);

            }

            

            return {

                ...obj, 

                NumericQty: qty, 

                NumericRevenue: revenueIncVat, 

                NumericNetPrice: unitPrice, 

                NumericCost: cost,

                YearMonth: obj.InvDate && obj.InvDate.length >= 7 ? obj.InvDate.substring(0, 7) : 'Unknown',

                // Map the channel here

                SaleChannelName: categorizeChannel(obj.SaleChannelName),

                ProductGroup: obj.deptname ? obj.deptname : (obj.ProductName ? 'Uncategorized' : 'Unknown')

            };

        };


        const parseCSV = (csvText) => {

            const lines = csvText.split(/\r?\n/);

            if (lines.length < 2) return [];

            const headers = lines[0].split(',').map(h => h.trim());

            const result = [];

            for (let i = 1; i < lines.length; i++) {

                if (!lines[i].trim()) continue;

                const currentline = lines[i].split(','); 

                const obj = {};

                for (let j = 0; j < headers.length; j++) {

                    obj[headers[j]] = currentline[j] ? currentline[j].trim() : '';

                }

                result.push(enrichData(obj));

            }

            return result;

        };

        const applyFilters = () => {

                       (!activeProductFilter || row.ProductName === activeProductFilter);

        const updateDashboard = () => {

            let totalRev = 0, totalCost = 0, totalQty = 0, returnQty = 0;

            filteredData.forEach(r => {

                totalRev += r.NumericRevenue;

                totalCost += r.NumericCost;

                if (r.NumericQty > 0) totalQty += r.NumericQty;

                if (r.NumericQty < 0) returnQty += Math.abs(r.NumericQty);

            });

            const profit = totalRev - totalCost;

            const margin = totalRev > 0 ? (profit / totalRev) * 100 : 0;

            const retRate = totalQty > 0 ? (returnQty / totalQty) * 100 : 0;


            elKpis.rev.innerText = formatCurr(totalRev);

            elKpis.profit.innerText = formatCurr(profit);

            elKpis.margin.innerText = `${margin.toFixed(1)}%`;

            elKpis.qty.innerText = formatNum(totalQty);

            elKpis.returnRate.innerText = `${retRate.toFixed(2)}%`;


            const tMap = {};

            filteredData.forEach(r => {

                const ym = r.YearMonth;

                if (ym === 'Unknown') return;

                tMap[ym].rev += r.NumericRevenue;

                tMap[ym].prof += (r.NumericRevenue - r.NumericCost);

            });

            const tData = Object.values(tMap).sort((a, b) => a.name.localeCompare(b.name));


            const bMap = {};

            filteredData.forEach(r => {

                const b = r.SaleChannelName || 'Unknown';

                if (!bMap[b]) bMap[b] = { name: b, rev: 0, prof: 0 };

                bMap[b].rev += r.NumericRevenue;

                bMap[b].prof += (r.NumericRevenue - r.NumericCost);

            });

            const bData = Object.values(bMap).map(b => ({

                ...b, margin: b.rev > 0 ? (b.prof / b.rev) * 100 : 0

            const pMap = {};

            filteredData.forEach(r => {

                const p = r.ProductName || 'Unknown';

                pMap[p].cost += r.NumericCost;

            });
